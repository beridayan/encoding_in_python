def encoding_code(code):
    final =""
    en = ""
    dici = {'s': 'L', 'b': 's', 'w': 'O', 'z': 'G', 'c': 'o', 'J': 'y', 'V': 't', 'P': 'w', 'B': 'f', 'Z': 'q', 'F': 'k', 'O': 'N', 'u': 'A', 'W': 'r', 'K': 'K', 'a': 'D', 'v': 'l', 'g': 'S', 'f': 'x', 'x': 'c', 'N': 'e', 'p': 'b', 'U': 'a', 'j': 'P', 'o': 'Q', 'i': 'I', 'M': 'd', 't': 'U', 'H': 'V', 'X': 'i', 'Y': 'T', 'R': 'H', 'h': 'X', 'L': 'z', 'G': 'F', 'A': 'W', 'm': 'n', 'T': 'u', 'l': 'B', 'C': 'Z', 'q': 'p', 'D': 'v', 'I': 'g', 'n': 'h', 'y': 'C', 'S': 'j', 'k': 'M', 'd': 'J', 'Q': 'E', 'e': 'Y', 'r': 'R', 'E': 'm'
    }
    new_dici = dict((v, k) for k, v in dici.items())#הופך את את המיליון שהפירוש הופך למילה הראשית והמילה הראשית הופכת לפירוש
    for x in range(len(code)):#הופך את כל המספרים לרווח
        if code[x].isdigit():
            final += " "
        else:
            final += code[x]
    revers = final[::-1]#הופך את הקוד
    for i in range(len(revers)):#מחליף את הקוד לאותיות מהמילון ואם יש אות שהיא לא במילון כמו פסיקים וסימן אזהרה אז הוא כותב אותם כמו שהם
        if revers[i] not in new_dici:
            en += revers[i]

        else:
            en += new_dici[revers[i]]
    def reversed_word(string):#הופך את הסדר של המילים
        finished = string.split(' ')
        finished.reverse()
        ff=' '.join(word for word in finished)
        return ff
    return reversed_word(en)
